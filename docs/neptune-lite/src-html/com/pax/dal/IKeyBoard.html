017package com.pax.dal;
018
019import com.pax.dal.entity.EKeyCode;
020
021public interface IKeyBoard {
022    /**
023     * @deprecated
024     * <div class="zh"> 检测键盘缓冲区中是否有尚未被取走的按键值 </div> <div class="en"> Check whether there are unread key values in
025     * keyboard buffer. </div>
026     * 
027     * @return <div class="zh">
028     *         <ul>
029     *         <li>true -- 键盘缓存区中仍然有键值</li>
030     *         <li>false -- 键盘缓存区中无键值</li>
031     *         </ul>
032     *         </div> <div class="en">
033     *         <ul>
034     *         <li>true -- There are still key values in buffer</li>
035     *         <li>false -- No key values in buffer</li>
036     *         </ul>
037     *         </div>
038     * 
039     */
040    @Deprecated
041    public boolean isHit();
042
043    /**
044     * @deprecated
045     * <div class="zh"> 清除当前键盘缓冲区中的所有未读取的按键 </div> <div class="en"> Clear all the key values which have not been read in
046     * current keyboard buffer. </div>
047     */
048    @Deprecated
049    public void clear();
050
051    /**
052     * @deprecated
053     * <div class="zh"> 读取键盘缓冲区中最早输入的一个键值。如果缓冲区为空，则等待按键输入 </div> <div class="en"> Read the first input key value
054     * from keyboard buffer. If the buffer is NULL, then waiting for the key pressing. </div>
055     * 
056     * @return {@link EKeyCode}
057     */
058    @Deprecated
059    public EKeyCode getKey();
060
061    /**
062     * @deprecated
063     * <div class="zh"> 设置按键板在按键时是否发声 </div> <div class="en"> Set whether the keypad should make a sound when the key is
064     * being pressed. </div>
065     * 
066     * @param isMute
067     *            <div class="zh">
068     *            <ul>
069     *            <li>true-不发声</li>
070     *            <li>false-发声</li>
071     *            </ul>
072     *            </div> <div class="en">
073     *            <ul>
074     *            <li>true -- Mute</li>
075     *            <li>false -- make a sound</li>
076     *            </ul>
077     *            </div>
078     * 
079     */
080    @Deprecated
081    public void setMute(boolean isMute);
082    
083    /**
084     * <div class="zh">设置按键背光。</div> <div class="en">Set keyboard backlight.</div>
085     *
086     * @param mode   <div class="zh">
087     *                  <ul>
088     *                     <li>0:关闭背光。</li>
089     *                     <li>1:默认模式。背光保持亮1分钟，1分钟后自动关闭。</li>
090     *                     <li>2:背光常亮。</li>
091     *                     <li>3:背光常暗。不会被刷卡，插卡，按键等事件激活背光。</li>
092     *                  </ul>
093     *               </div>
094     *               <div class="en">
095     *                  <ul>
096     *                     <li>0:Turn off the backlight.</li>
097     *                     <li>1:Default mode.Keep the backlight on for 1 minute, then turn it off automatically after 1 minute.</li>
098     *                     <li>2:The backlight is always bright.</li>
099     *                     <li>3:The backlight is always dark.Backlight will not be activated by card swiping, card inserting, keystrokes, etc.</li>
100     *                  </ul>
101     *               </div>
102     * @param volume <div class="zh">范围是1~3。默认3最亮。</div> <div class="en">The range is 1 to 3.Default 3 is brightest.</div>
103     * @throws Exception
104     * @since V3.19.00
105     */
106    void setLight(byte mode, byte volume) throws Exception;
107
108    /**
109     * @deprecated
110     * <div class="zh">设置键盘按键音量值。</div> <div class="en">Set keyboard key volume value.</div>
111     *
112     * @param volumes   <div class="zh">音量值（0~50）</div>
113     *               <div class="en">Volume value (0~50)</div>
114     * @throws Exception
115     * @since V3.29.00
116     */
117    void setVolume(int volumes) throws Exception;
118}
