017package com.pax.dal.entity;
018
019/**
020 *  <div class="zh">
021     *  串口定义
022     * </div>
023     * <div class="en">
024     * uart port
025     * </div>
026 * @author Steven.W
027 */
028public enum EUartPort  {
029        
030        /**
031         * com1
032         */
033    COM1(0),
034    /**
035     * com2
036     */
037    COM2(1),
038    /**
039     * wireless
040     */
041    WIRELESS(2),
042    /**
043     * pinpad
044     */
045    PINPAD(3),
046    /**
047     * modem
048     */
049    MODEM(4),
050    /**
051     * Wi-Fi
052     */
053    WIFI(5),
054    /**
055     * visit FTDI USB convert serial port
056     */
057    FTDIUSB(10),
058    /**
059     * USB device mode port
060     */
061    USBDEV(11),
062
063    /**
064     * USB host mode port
065     */
066    @Deprecated
067    USBHOST(12),
068    /**
069     * USB CCID device mode port
070     */
071    @Deprecated
072    USBCCID(13),
073    
074    /**
075     * Internal communication channel between Android system and security module
076     */
077    COM19(19),
078    
079    /**
080     * USB convert serial port. Same as {@link #FTDIUSB}.
081     */
082    USBCOM10(10),
083
084    /**
085     * USB convert serial port.
086     * @since V3.16.00
087     */
088    USBCOM30(30),
089
090    /**
091     * <div class="zh">USB设备模式端口。仅当授权功能被禁用时才能使用。</div>
092     * <div class="en">USB device mode port. It can be used only when the authorization function is disabled.</div>
093     * @since V3.22.00
094     */
095    USBDEV50(50)
096    ;
097
098    private byte channel;
099
100    private EUartPort(int channel) {
101        this.channel = (byte) channel;
102    }
103
104  
105    public byte getChannel() {
106        return channel;
107    }
108    
109    public static EUartPort valuesOf(int channel) {
110        for (EUartPort up : EUartPort.values()) {
111            if (up.getChannel() == (byte) channel)
112                return up;
113        }
114        return null;
115    }
116
117}
