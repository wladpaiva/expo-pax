017package com.pax.dal.entity;
018
019public class WriteKEKOutput {
020
021
022    /**
023     * <div> Output the PED secret number, 16 bytes </div>
024     */
025    private byte[] pedSnOut;
026
027    /**
028     * <div> Output the RKI secret number, 16 bytes</div>
029     */
030    private byte[] rkiSnOut ;
031
032    public WriteKEKOutput(byte[] pedSnOut, byte[] rkiSnOut) {
033        this.pedSnOut = pedSnOut;
034        this.rkiSnOut = rkiSnOut;
035    }
036
037    /**
038     * @return <div> PED secret number, 16 bytes</div>
039     */
040    public byte[] getPedSnOut() {
041        return pedSnOut;
042    }
043
044    /**
045     * @return <div> RKI secret number, 16 bytes</div>
046     */
047    public byte[] getRkiSnOut() {
048        return rkiSnOut;
049    }
050}
