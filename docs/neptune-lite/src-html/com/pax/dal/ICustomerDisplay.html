016package com.pax.dal;
017
018import android.graphics.Bitmap;
019
020import com.pax.dal.entity.CustomerDisplayInfo;
021import com.pax.dal.exceptions.CustomerDisplayDevException;
022
023/**
024 * <div class="zh">顾客显示屏操作模块。</div>
025 * <div class="en">Customer display operation module.</div>
026 * @since V3.25.00
027 */
028public interface ICustomerDisplay {
029
030    /**
031     * <div class="zh"> 获取顾客显示屏属性。</div> <div class="en"> Get customer display properties. </div>
032     *
033     * @return {@link CustomerDisplayInfo}
034     * @throws CustomerDisplayDevException CustomerDisplayDevException
035     * @since V3.25.00
036     */
037    CustomerDisplayInfo getProperty() throws CustomerDisplayDevException;
038
039    /**
040     * <div class="zh"> 设置顾客显示屏亮度。</div> <div class="en"> Set customer display brightness. </div>
041     *
042     * @param level <div class="zh"> 亮度。范围 0~10，0表示屏幕全黑。默认值6。</div>
043     *              <div class="en"> Brightness. The range is 0 ~ 10, 0 means the screen is completely black. The default value is 6. </div>
044     * @throws CustomerDisplayDevException CustomerDisplayDevException
045     * @since V3.25.00
046     */
047    void setBrightness(int level) throws CustomerDisplayDevException;
048
049    /**
050     * <div class="zh"> 设置顾客显示屏的背景图片。</div> <div class="en">Set the background image of the customer display. </div>
051     *
052     * @param x <div class="zh"> x坐标。</div> <div class="en"> X coordinates.</div>
053     * @param y <div class="zh"> y坐标。</div> <div class="en"> Y coordinates.</div>
054     * @param background <div class="zh"> 背景图像。如果图片像素超过屏幕，将按原图片比例进行缩放。只支持png格式。</div>
055     *                   <div class="en"> Background image. If the picture pixels exceed the screen, it will be scaled to the original picture scale.
056     *                   Only PNG format is supported.</div>
057     * @throws CustomerDisplayDevException CustomerDisplayDevException
058     * @since V3.25.00
059     */
060    void setBackgroundImage(int x, int y, Bitmap background) throws CustomerDisplayDevException;
061
062    /**
063     * <div class="zh"> 设置顾客显示屏的纯色背景。</div> <div class="en"> Set the solid background of the customer display.</div>
064     *
065     * @param color <div class="zh">
066     *              
067     *               <ul>
068     *                  <li>0: 红色。</li>
069     *                  <li>1: 绿色。</li>
070     *                  <li>2: 蓝色。</li>
071     *                  <li>3: 黑色。</li>
072     *                  <li>4: 白色。</li>
073     *                  <li>5: 黄色。</li>
074     *                  <li>6: 灰色。</li>
075     *               </ul>
076     *              </div> 
077     *              <div class="en">
078     *               <ul>
079     *                  <li>0: RED.</li>
080     *                  <li>1: GREEN.</li>
081     *                  <li>2: BLUE.</li>
082     *                  <li>3: BLACK.</li>
083     *                  <li>4: WHITE.</li>
084     *                  <li>5: YELLOW.</li>
085     *                  <li>6: GRAY.</li>
086     *               </ul>
087     *              </div>
088     * @throws CustomerDisplayDevException CustomerDisplayDevException
089     * @since V3.25.00
090     */
091    void setBackgroundSolid(int color) throws CustomerDisplayDevException;
092
093    /**
094     * <div class="zh"> 设置顾客显示屏文字。</div> <div class="en"> Set the customer display text. </div>
095     *
096     * @param x <div class="zh"> x坐标。</div> <div class="en"> X coordinates.</div>
097     * @param y <div class="zh"> y坐标。</div> <div class="en"> Y coordinates.</div>
098     * @param text <div class="zh"> 要显示的文字。</div> <div class="en"> The text to display. </div>
099     * @param charset <div class="zh"> 文字的编码格式。为NULL表示UTF-8。</div> <div class="en"> The encoding format of the text. Null means UTF-8. </div>
100     * @throws CustomerDisplayDevException CustomerDisplayDevException
101     * @since V3.25.00
102     */
103    void setText(int x, int y, String text, String charset) throws CustomerDisplayDevException;
104
105    /**
106     * <div class="zh"> 清除客显屏指定区域。</div> <div class="en"> Clear the designated area of the Customer screen.</div>
107     *
108     * @param x      <div class="zh"> x坐标。</div> <div class="en"> X coordinates.</div>
109     * @param y      <div class="zh"> y坐标。</div> <div class="en"> Y coordinates.</div>
110     * @param width  <div class="zh"> 清除宽度。 </div> <div class="en"> Clearance width.</div>
111     * @param height <div class="zh"> 清除高度。</div> <div class="en"> Clearance height.</div>
112     * @throws CustomerDisplayDevException CustomerDisplayDevException
113     * @since V3.25.00
114     */
115    void clear(int x, int y, int width, int height) throws CustomerDisplayDevException;
116
117    /**
118     * <div class="zh"> 设置顾客显示屏的字体。</div> <div class="en"> Set the font of the customer display.</div>
119     *
120     * @param style  <div class="zh">
121     *                <ul>
122     *                 <li>0: 正常。</li>
123     *                 <li>1: 加粗。</li>
124     *                 <li>2: 斜体。</li>
125     *                 <li>3: 斜体加粗。</li>
126     *                </ul>
127     *               </div>
128     *               <div class="en">
129     *                <ul>
130     *                 <li>0: normal.</li>
131     *                 <li>1: bold.</li>
132     *                 <li>2: italic.</li>
133     *                 <li>3: bold italic.</li>
134     *                </ul>
135     *               </div>
136     * @param height <div class="zh"> 字体高度。单位: 像素。</div> <div class="en"> Font height. Unit: pixel.</div>
137     * @throws CustomerDisplayDevException CustomerDisplayDevException
138     * @since V3.25.00
139     */
140    void setFont(int style, int height) throws CustomerDisplayDevException;
141
142    /**
143     * <div class="zh"> 启动顾客显示屏的签名板。 </div> <div class="en"> Start the signature board on the customer display. </div>
144     *
145     * @param timeout <div class="zh"> 超时时间。单位: 毫秒。</div> <div class="en"> Timeout. unit: millisecond.</div>
146     * @throws CustomerDisplayDevException CustomerDisplayDevException
147     * @since V3.25.00
148     */
149    void startSignBoard(int timeout) throws CustomerDisplayDevException;
150
151    /**
152     * <div class="zh"> 获取顾客显示屏的签名图像。</div> <div class="en"> Get the signature image of customer display. </div>
153     *
154     * @return <div class="zh"> 签名图像。 </div> <div class="en"> Signature image. </div>
155     * @throws CustomerDisplayDevException CustomerDisplayDevException
156     * @since V3.25.00
157     */
158    Bitmap getSignBoardImage() throws CustomerDisplayDevException;
159
160}
