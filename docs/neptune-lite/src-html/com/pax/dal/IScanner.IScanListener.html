017package com.pax.dal;
018
019import com.pax.dal.entity.EUartPort;
020import com.pax.dal.entity.ScanResult;
021
022 public interface IScanner {
023
024    public interface IScanListener {
025        /**
026         * <div class="zh"> 读取成功回调 </div> <div class="en"> the underlying called when read success </div>
027         * 
028         * @param scanResult {@link ScanResult}
029         */
030        public void onRead(ScanResult scanResult);
031
032        /**
033         * <div class="zh"> 扫描结束回调 </div> <div class="en"> the underlying called when finish scanning </div>
034         * 
035         */
036        public void onFinish();
037
038        /**
039         * <div class="zh"> 取消扫描回调 </div> <div class="en"> the underlying called when cancelled </div>
040         * 
041         */
042        public void onCancel();
043    }
044
045    /**
046     * <div class="zh"> 打开扫码 </div> <div class="en"> open scanner </div>
047     * 
048     * @return <div class="zh">
049     *         <ul>
050     *         <li>true -- 打开成功</li>
051     *         <li>false -- 打开失败</li>
052     *         </ul>
053     *         </div> <div class="en">
054     *         <ul>
055     *         <li>true -- open success</li>
056     *         <li>false -- failure</li>
057     *         </ul>
058     *         </div>
059     * 
060     * @return
061     */
062    public boolean open();
063
064    /**
065     * <div class="zh"> 开始扫描 </div> <div class="en"> begin scan </div>
066     * 
067     * @param listener
068     *            {@link IScanListener}
069     */
070    public void start(IScanListener listener);
071
072    /**
073     * <div class="zh"> 关闭摄像头 </div> <div class="en"> close scanner </div>
074     * 
075     */
076    public void close();
077
078    /**
079     * <div class="zh"> 设置超时时间</div> <div class="en">set timeout </div>
080     * 
081     * @param timeOut
082     *            <div class="zh"> 超时时间 </div> <div class="en"> timeout </div>
083     */
084    public void setTimeOut(int timeOut);
085
086    /**
087     * <div class="zh"> 设置最大扫码次数</div> <div class="en">Set the maximum number of scan </div>
088     * 
089     * @param times
090     *            <div class="zh"> 最大扫码次数,0代表连续扫码</div> <div class="en">Maximum scan times,0 represents continuous
091     *            scan.</div>
092     * @return <div class="zh"> true:设置成功; false:设置失败;</div> <div class="en">true:set successful false:set failed </div>
093     * @since V2.02.00
094     */
095    boolean setContinuousTimes(int times);
096
097    /**
098     * <div class="zh"> 设置扫码间隔</div> <div class="en">Set scan interval </div>
099     * 
100     * @param interval
101     *            <div class="zh"> 扫码间隔时间(ms)</div> <div class="en">Scan interval time (ms)</div>
102     * @since V2.02.00
103     */
104    void setContinuousInterval(int interval);
105    
106    /**
107     * <div class="zh"> 设置外置扫码模块的连接端口</div> <div class="en"> Set connection port of the external scanner</div>
108     * 
109     * @param port
110     *            {@link EUartPort}
111     * @since V3.02.00
112     */
113    void setPort(EUartPort port);
114    
115    /**
116     * <div class="zh">设置扫码时是否开启闪光灯。</div> <div class="en">Set whether flash is turned on when scanning.</div>
117     * 
118     * @param isOn <div class="zh">
119     * <ul>
120     * <li>true:开启。</li>
121     * <li>false:关闭。</li>
122     * </ul></div> 
123     * <div class="en">
124     * <ul>
125     * <li>true:Turn on.</li>
126     * <li>false:Turn off.</li>
127     * </ul></div>
128     * 
129     * @return <div class="zh">设置结果。
130     * <ul>
131     * <li>true:成功。</li>
132     * <li>false:失败。</li>
133     * </ul></div> 
134     * <div class="en">The result of setting.
135     * <ul>
136     * <li>true:success.</li>
137     * <li>false:failure.</li>
138     * </ul></div>
139     * @since V3.11.00
140     */
141    boolean setFlashOn(boolean isOn);
142
143    /**
144     * <div class="zh">停止扫码。要和{@link #start(IScanListener)}成对使用。</div>
145     * <div class="en">Stop scanning. Use it in pairs with {@link #start(IScanListener)}.</div>
146     * @since V3.25.00
147     */
148    void stop();
149
150    /**
151     * <div class="zh">设置扫码类型。</div> <div class="en">Set the scan type.</div>
152     *
153     * @param type <div class="zh">扫码类型。
154     *              <ul>
155     *               <li>0:Zxing</li>
156     *               <li>1:Cortex</li>
157     *               <li>2:ScanIt</li>
158     *               <li>3:ScanS</li>
159     *               <li>4:LiveScan</li>
160     *              </ul>
161     *             </div>
162     *             <div class="en">The scan type.
163     *              <ul>
164     *               <li>0:Zxing</li>
165     *               <li>1:Cortex</li>
166     *               <li>2:ScanIt</li>
167     *               <li>3:ScanS</li>
168     *               <li>4:LiveScan</li>
169     *              </ul>
170     *             </div>
171     * @return <div class="zh">结果。
172     *          <ul>
173     *           <li>true:成功。</li>
174     *           <li>false:失败。</li>
175     *          </ul>
176     *         </div>
177     *         <div class="en">The result.
178     *          <ul>
179     *           <li>true:success.</li>
180     *           <li>false:failure.</li>
181     *          </ul>
182     *         </div>
183     * @since V3.25.00
184     */
185    boolean setScannerType(int type);
186}
