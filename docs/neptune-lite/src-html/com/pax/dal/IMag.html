017package com.pax.dal;
018
019import com.pax.dal.entity.TrackData;
020import com.pax.dal.exceptions.MagDevException;
021
022 /**
023  *  <div class="zh">关于磁条卡的API。 如果开启权限检查，需在AndroidManifest下添加权限:"com.pax.permission.MAGCARD"</div>
024  *  <div class="en">API for Magnetic stripe cards. If permission check is enabled, you need to add permission under AndroidManifest: "com.pax.permission.MAGCARD".</div>
025  */
026public interface IMag {
027    /**
028     * <div class="zh"> 打开读卡器 </div> <div class="en"> Open Magnetic Stripe Card Reader </div>
029     * 
030     * @throws MagDevException
031     */
032    public void open() throws MagDevException;
033
034    /**
035     * <div class="zh"> 复位磁头，并清除磁卡缓冲区数据 </div> <div class="en"> Reset magnetic stripe card reader, and clear buffer of
036     * magnetic stripe card. </div>
037     * 
038     * @throws MagDevException
039     */
040    public void reset() throws MagDevException;
041
042    /**
043     * <div class="zh"> 检查是否刷过卡 </div> <div class="en"> Check whether a card is swiped </div>
044     * 
045     * @return <div class="zh"> *
046     *         <ul>
047     *         <li>true -有刷卡</li>
048     *         <li>false -没有刷卡</li>
049     *         </ul>
050     *         </div> <div class="en">
051     *         <ul>
052     *         <li>true -Card swiped</li>
053     *         <li>false -No card swiping</li>
054     *         </ul>
055     *         </div>
056     * 
057     * @throws MagDevException
058     */
059    public boolean isSwiped() throws MagDevException;
060
061    /**
062     * <div class="zh"> 读取磁卡缓冲区的1、2、3磁道的数据 </div> <div class="en"> Read data of track 1, 2, 3 from magnetic stripe card
063     * buffer. </div>
064     * 
065     * @return {@link TrackData}
066     * @throws MagDevException
067     */
068    public TrackData read() throws MagDevException;
069
070    /**
071     * <div class="zh"> 关闭读卡器 </div> <div class="en"> Close Magnetic Stripe Card Reader </div>
072     * 
073     * @throws MagDevException
074     */
075    public void close() throws MagDevException;
076    
077    /**
078     * <div class="zh"> 读取磁卡缓冲区的1、2、3磁道或JIS-II磁道的数据。 </div> <div class="en"> Read data of track 1, 2, 3 or
079     * JIS-II from magnetic stripe card buffer.</div>
080     * 
081     * @return {@link TrackData}
082     * 
083     * @throws MagDevException
084     * @since V3.08.00
085     */
086    TrackData readExt() throws MagDevException;
087    
088    /**
089     * <div class="zh">磁卡功能设置：是否返回磁卡错误码。设置有效周期：设置后直到机器重启或下一次重新设置新的值。</div>
090     * <div class="en">Magnetic card function Settings: whether to return magnetic card error code. Set duration: set until the machine 
091     *  restarts or reset the new value the next time. </div>
092     * 
093     * @param flag <div class="zh">
094     * <ul>
095     *  <li>0:{@link #read()}过滤磁卡错误码(默认模式)。</li>
096     *  <li>1:{@link #read()}返回此卡错误码。</li>
097     * </ul></div> <div class="en">
098     * <ul>
099     *  <li>0:{@link #read()}Filter card error code (default mode).</li>
100     *  <li>1:{@link #read()}Returns the error code for this card.</li>
101     * </ul></div>
102     * 
103     * @throws MagDevException
104     * @since V3.16.00
105     */
106    void setup(byte flag) throws MagDevException;
107}
